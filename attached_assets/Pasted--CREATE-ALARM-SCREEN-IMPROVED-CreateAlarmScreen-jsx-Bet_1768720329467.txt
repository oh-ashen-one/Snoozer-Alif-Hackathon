              </button>

                AM

            PROOF OF WAKE

                  <button
                    key={amt}
                    style={{
                      ...styles.amountButton,
                      ...(amount === amt ? styles.amountButtonActive : {}),

                  >

                  </button>

                <span>ğŸ’¡</span>

            ESCALATION
        â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â• */}

            value={wakeRecheck}
            onChange={() => setWakeRecheck(!wakeRecheck)}
          />

            icon="ğŸ“¹"
            title="Shame Video"
            subtitle="Plays at max volume if you snooze"
            value={shameVideo}
            onChange={() => setShameVideo(!shameVideo)}

        <div style={styles.summaryCard}>
          <h3 style={styles.summaryTitle}>Summary</h3>

              {hour}:{minute.toString().padStart(2, '0')} {isPM ? 'PM' : 'AM'}
            </span>
          </div>
          <div style={styles.summaryRow}>
            <span style={styles.summaryLabel}>Repeat</span>
            <span style={styles.summaryValue}>{getRepeatLabel()}</span>
          </div>
          <div style={styles.summaryRow}>
            <span style={styles.summaryLabel}>Dismiss</span>
            <span style={styles.summaryValue}>{selectedProofData?.name}</span>

            </span>
          </div>
        </div>
      </div>

  return (
    <div style={styles.toggleRow}>
      <span style={styles.toggleIcon}>{icon}</span>
      <div style={styles.toggleInfo}>
        <span style={styles.toggleTitle}>{title}</span>
        <span style={styles.toggleSubtitle}>{subtitle}</span>
      </div>
      <button
        style={{
          ...styles.toggle,
          backgroundColor: value ? '#22C55E' : '#292524',

        onClick={onChange}

        <div style={{
          ...styles.toggleKnob,
          transform: value ? 'translateX(20px)' : 'translateX(0)',

    </div>
